import { useState } from "react";

export const Counter = () => {
    const [count, setCount] = useState(0);
    return (
        <div>
            <button
                onClick={() => setCount(count + 1)}
                className="bg-blue-500 hover:bg-red-500 duration-1000 p-4"
            >
                Clicked {count} times
            </button>
        </div>
    );
};

<Counter />
